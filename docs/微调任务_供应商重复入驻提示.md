```python
async def test_supplier_register_duplicate_license_returns_friendly_message(client, db):
    # 第一次注册:成功
    resp1 = await client.post("/api/v1/auth/register/supplier", json={
        "email": "owner@xx.com", "name": "owner",
        "password": "abc12345", "company_name": "XX 公司",
        "business_license_no": "9144XXXXMA001",
    })
    assert resp1.status_code == 200

    # 第二次同执照号:返回 409 + 指定文案
    resp2 = await client.post("/api/v1/auth/register/supplier", json={
        "email": "another@xx.com", "name": "another",
        "password": "abc12345", "company_name": "XX 公司",
        "business_license_no": "9144XXXXMA001",
    })
    assert resp2.status_code == 409
    body = resp2.json()
    assert "该供应商已在平台注册" in body["message"]
    assert "联系企业管理员" in body["message"]

    # 断言:DB 中 user 二无任何数据
    from sqlalchemy import select
    from app.db.models.user import User
    row = await db.execute(select(User).where(User.email == "another@xx.com"))
    assert row.scalar_one_or_none() is None
```
